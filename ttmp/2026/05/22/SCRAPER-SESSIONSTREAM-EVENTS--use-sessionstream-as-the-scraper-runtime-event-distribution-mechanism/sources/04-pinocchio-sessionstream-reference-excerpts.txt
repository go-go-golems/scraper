    19	func (e *Engine) handleStartInference(ctx context.Context, cmd sessionstream.Command, _ *sessionstream.Session, pub sessionstream.EventPublisher) error {
    20		payload, ok := cmd.Payload.(*chatappv1.StartInferenceCommand)
    21		if !ok || payload == nil {
    22			return fmt.Errorf("start inference payload must be %T, got %T", &chatappv1.StartInferenceCommand{}, cmd.Payload)
    23		}
    24		pending := e.takePendingRequest(strings.TrimSpace(payload.GetRequestId()))
    25		prompt := strings.TrimSpace(pending.Prompt)
    26		if prompt == "" {
    27			prompt = strings.TrimSpace(payload.GetPrompt())
    28		}
    29		if prompt == "" {
    30			prompt = "Explain evtstream"
    31		}
    32		messageID := e.nextMessageID()
    33		userMessageID := messageID + "-user"
    34		if err := e.publish(ctx, cmd.SessionId, pub, EventUserMessageAccepted, &chatappv1.ChatUserMessageAccepted{MessageId: userMessageID, Role: "user", Text: prompt, Content: prompt, Status: "accepted"}); err != nil {
    35			return err
    36		}
    37		runCtx, cancel := context.WithCancel(publishContext(ctx))
    38		run := &activeRun{messageID: messageID, cancel: cancel, done: make(chan struct{})}
    39		if previous := e.swapRun(cmd.SessionId, run); previous != nil {
    40			previous.cancel()
    41			<-previous.done
    42		}
    43		go e.runPrompt(runCtx, cmd.SessionId, messageID, pending, prompt, pub, run.done)
    44		return nil
    45	}

    79				e.publishRunFailed(publishContext(ctx), sid, pub, messageID, err.Error())
    80				return
    81			}
    82			eventSink = wrapped
    83		}
    84		sink, ok := baseSink.(*runtimeEventSink)
    85		if !ok {
    86			e.publishRunFailed(publishContext(ctx), sid, pub, messageID, "internal runtime sink type assertion failed")
    87			return
    88		}
    89		sess := gepsession.NewSessionWithID(string(sid))
    90		sess.Builder = &enginebuilder.Builder{
    91			Base:       runtime.Engine,
    92			EventSinks: []gepevents.EventSink{eventSink},
    93		}
    94	
    95		if pending.InitialTurn != nil {
    96			sess.Append(pending.InitialTurn.Clone())
    97		} else {
    98			// Load conversation history: the last persisted turn contains the full
    99			// conversation as an accumulator. AppendNewTurnFromUserPrompt will clone
   100			// it and add the new user block, giving the LLM the full context.
   101			if e.turnStore != nil {
   102				snapshot, err := e.turnStore.LoadLatestTurn(ctx, string(sid), "final")
   103				if err != nil {
   104					e.publishRunFailed(publishContext(ctx), sid, pub, messageID, fmt.Sprintf("load conversation history: %v", err))
   105					return
   106				}
   107				if snapshot != nil {
   108					turn, err := serde.FromYAML([]byte(snapshot.Payload))
   109					if err != nil {
   110						e.publishRunFailed(publishContext(ctx), sid, pub, messageID, fmt.Sprintf("decode conversation history: %v", err))
   111						return
   112					}
   113					if turn == nil {
   114						e.publishRunFailed(publishContext(ctx), sid, pub, messageID, "decode conversation history: empty turn")
   115						return
   116					}
   117					sess.Append(turn)
   118				}
   119			}
   120	
   121			_, err := sess.AppendNewTurnFromUserPrompt(prompt)
   122			if err != nil {
   123				e.publishRunFailed(publishContext(ctx), sid, pub, messageID, err.Error())
   124				return
   125			}
   126		}
   127		assistantBlockOffset := countAssistantBlocks(sess.Latest())
   128		handle, err := sess.StartInference(ctx)
   129		if err != nil {
   130			e.publishRunFailed(publishContext(ctx), sid, pub, messageID, err.Error())
   131			return
   132		}
   133		output, err := handle.Wait()
   134		if err != nil {
   135			if !sink.IsTerminal() {
   136				if errors.Is(err, context.Canceled) || ctx.Err() != nil {
   137					_ = sink.finishActiveTextSegment("stopped", "stopped", "")
   138					_ = e.publish(publishContext(ctx), sid, pub, EventChatRunStopped, &chatappv1.ChatRunStopped{MessageId: messageID, Status: "stopped", Correlation: runCorrelationInfo(sid, messageID)})
   139					return
   140				}
   141				_ = sink.finishActiveTextSegment("failed", "error", "")
   142				if isMaxIterationsError(err) {
   143					_ = e.publish(publishContext(ctx), sid, pub, EventChatTextSegmentFinished, &chatappv1.ChatTextSegmentFinished{MessageId: runtimeWarningMessageID(messageID), Role: "warning", Prompt: prompt, Text: maxIterationsWarningText(err), Content: maxIterationsWarningText(err), Status: "finished", Streaming: false, Final: true})
   144				}
   145				e.publishRunFailed(publishContext(ctx), sid, pub, messageID, err.Error())
   146			}
   147			return
   148		}
   149		if pending.OnFinalTurn != nil && output != nil {
   150			pending.OnFinalTurn(output.Clone())
   151		}
   152		if sink.IsTerminal() {
   153			return
   154		}
   155		if sink.HasActiveTextSegment() {
   156			_ = sink.finishActiveTextSegment("finished", "stop", "")
   157		} else if !sink.HasTextSegment() {
   158			_ = e.publishFallbackAssistantText(publishContext(ctx), sid, pub, messageID, prompt, output, assistantBlockOffset)
   159		}
   160		_ = e.publish(publishContext(ctx), sid, pub, EventChatRunFinished, &chatappv1.ChatRunFinished{MessageId: messageID, Status: "finished", Correlation: runCorrelationInfo(sid, messageID)})
   161	}

   212		for _, block := range turn.Blocks {
   213			if block.Role != turns.RoleAssistant {
   214				continue
   215			}
