   273	// Run starts the configured bus consumer if an event bus is present.
   274	func (h *Hub) Run(ctx context.Context) error {
   275		if h == nil {
   276			return fmt.Errorf("hub is nil")
   277		}
   278		if h.bus == nil {
   279			return nil
   280		}
   281		if ctx == nil {
   282			return fmt.Errorf("ctx is nil")
   283		}
   284	
   285		h.runMu.Lock()
   286		defer h.runMu.Unlock()
   287		if h.runCancel != nil {
   288			return nil
   289		}
   290		runCtx, cancel := context.WithCancel(ctx)
   291		consumer := newEventConsumer(h)
   292		if err := consumer.start(runCtx); err != nil {
   293			cancel()
   294			return err
   295		}
   296		h.runCancel = cancel
   297		h.consumer = consumer
   298		return nil
   299	}

   301	// Shutdown stops the active bus consumer.
   302	func (h *Hub) Shutdown(ctx context.Context) error {
   303		if h == nil {
   304			return fmt.Errorf("hub is nil")
   305		}
   306		h.runMu.Lock()
   307		cancel := h.runCancel
   308		consumer := h.consumer
   309		h.runCancel = nil
   310		h.consumer = nil
   311		h.runMu.Unlock()
   312		if cancel == nil || consumer == nil {
   313			return nil
   314		}
   315		cancel()
   316		return consumer.wait(ctx)
   317	}

   342	func (p localEventPublisher) Publish(ctx context.Context, ev Event) error {
   343		if p.hub == nil {
   344			return fmt.Errorf("hub is nil")
   345		}
   346		if ev.SessionId == "" {
   347			return fmt.Errorf("event %q missing session id", ev.Name)
   348		}
   349		if ev.Payload == nil {
   350			return fmt.Errorf("event %q payload is nil", ev.Name)
   351		}
   352		if err := p.hub.validatePayloadType(p.hub.reg.events, "event", ev.Name, ev.Payload); err != nil {
   353			return err
   354		}
   355		ord, err := p.hub.nextLocalOrdinal(ctx, ev.SessionId)
   356		if err != nil {
   357			return err
   358		}
   359		ev.Ordinal = ord
   360		_, err = p.hub.projectAndApply(ctx, ev)
   361		return err
   362	}

   390	func (h *Hub) RebuildTimeline(ctx context.Context, sid SessionId, from uint64) error {
   391		if h == nil {
   392			return fmt.Errorf("hub is nil")
   393		}
   394		if sid == "" {
   395			return fmt.Errorf("session id is empty")
   396		}
   397		if h.timelineProjection == nil {
   398			return fmt.Errorf("timeline projection is not registered")
   399		}
   400		eventStore, ok := h.store.(EventStore)
   401		if !ok {
   402			return fmt.Errorf("hydration store does not support event replay")
   403		}
   404		for {
   405			events, err := eventStore.Events(ctx, sid, from, 100)
   406			if err != nil {
   407				h.reportError(ctx, ErrorRecord{Kind: ErrorKindStore, SessionId: sid, Ordinal: from, Err: err})
   408				return err
   409			}
   410			if len(events) == 0 {
   411				return nil
   412			}
   413			for _, ev := range events {
   414				if err := h.rebuildTimelineEvent(ctx, ev); err != nil {
   415					return err
   416				}
   417				from = ev.Ordinal
   418			}
   419		}
   420	}

   422	func (h *Hub) rebuildTimelineEvent(ctx context.Context, ev Event) error {
   423		rec := PipelineRecord{Mode: PipelineModeRebuild, SessionId: ev.SessionId, Ordinal: ev.Ordinal, EventName: ev.Name, Event: cloneEvent(ev)}
   424		defer func() { h.observePipeline(ctx, rec) }()
   425	
   426		sess, err := h.sessions.GetOrCreate(ctx, ev.SessionId)
   427		if err != nil {
   428			rec.SessionErr = err
   429			return err
   430		}
   431		view, err := h.store.View(ctx, ev.SessionId)
   432		if err != nil {
   433			rec.ViewErr = err
   434			h.reportError(ctx, ErrorRecord{Kind: ErrorKindStore, SessionId: ev.SessionId, Ordinal: ev.Ordinal, EventName: ev.Name, Err: err})
   435			return err
   436		}
   437		rec.ViewOrdinal = view.Ordinal()
   438		entities, err := h.timelineProjection.Project(ctx, ev, sess, view)
   439		rec.TimelineEntities = cloneTimelineEntities(entities)
   440		rec.TimelineProjectionErr = err
   441		if err != nil {
   442			h.reportError(ctx, ErrorRecord{Kind: ErrorKindTimelineProjection, SessionId: ev.SessionId, Ordinal: ev.Ordinal, EventName: ev.Name, Err: err})
   443			return err
   444		}
   445		if err := h.store.Apply(ctx, ev.SessionId, ev.Ordinal, entities); err != nil {
   446			rec.AppliedEntities = cloneTimelineEntities(entities)
   447			rec.ApplyErr = err
   448			h.reportError(ctx, ErrorRecord{Kind: ErrorKindStore, SessionId: ev.SessionId, Ordinal: ev.Ordinal, EventName: ev.Name, Err: err})
   449			return err
   450		}
   451		rec.AppliedEntities = cloneTimelineEntities(entities)
   452		if projectionStore, ok := h.store.(ProjectionCursorStore); ok {
   453			if err := projectionStore.AdvanceProjectionCursor(ctx, TimelineProjectorName, ev.SessionId, ev.Ordinal); err != nil {
   454				rec.CursorErr = err
   455				h.reportError(ctx, ErrorRecord{Kind: ErrorKindStore, SessionId: ev.SessionId, Ordinal: ev.Ordinal, EventName: ev.Name, Err: err})
   456				return err
   457			}
   458			rec.TimelineCursorAdvanced = true
   459		}
   460		return nil
   461	}

   463	func (h *Hub) projectAndApply(ctx context.Context, ev Event) ([]UIEvent, error) {
   464		rec := PipelineRecord{Mode: PipelineModeLive, SessionId: ev.SessionId, Ordinal: ev.Ordinal, EventName: ev.Name, Event: cloneEvent(ev)}
   465		defer func() { h.observePipeline(ctx, rec) }()
   466	
   467		if eventStore, ok := h.store.(EventStore); ok {
   468			if err := eventStore.AppendEvent(ctx, ev); err != nil {
   469				rec.AppendErr = err
   470				h.reportError(ctx, ErrorRecord{Kind: ErrorKindStore, SessionId: ev.SessionId, Ordinal: ev.Ordinal, EventName: ev.Name, Err: err})
   471				return nil, err
   472			}
   473			rec.EventAppended = true
   474		}
   475	
   476		sess, err := h.sessions.GetOrCreate(ctx, ev.SessionId)
   477		if err != nil {
   478			rec.SessionErr = err
   479			return nil, err
   480		}
   481		view, err := h.store.View(ctx, ev.SessionId)
   482		if err != nil {
   483			rec.ViewErr = err
   484			return nil, err
   485		}
   486		rec.ViewOrdinal = view.Ordinal()
   487	
   488		var (
   489			uiEvents []UIEvent
   490			entities []TimelineEntity
   491			uiErr    error
   492			tlErr    error
   493		)
   494		if h.uiProjection != nil {
   495			uiEvents, uiErr = h.uiProjection.Project(ctx, ev, sess, view)
   496			rec.UIEvents = cloneUIEvents(uiEvents)
   497			rec.UIProjectionErr = uiErr
   498		}
   499		if h.timelineProjection != nil {
   500			entities, tlErr = h.timelineProjection.Project(ctx, ev, sess, view)
   501			rec.TimelineEntities = cloneTimelineEntities(entities)
   502			rec.TimelineProjectionErr = tlErr
   503		}
   504	
   505		if uiErr != nil {
   506			h.reportError(ctx, ErrorRecord{Kind: ErrorKindUIProjection, SessionId: ev.SessionId, Ordinal: ev.Ordinal, EventName: ev.Name, Err: uiErr})
   507			if h.projectionPolicies.UI == ProjectionErrorPolicyFail {
   508				return nil, uiErr
   509			}
   510		}
   511		if tlErr != nil {
   512			h.reportError(ctx, ErrorRecord{Kind: ErrorKindTimelineProjection, SessionId: ev.SessionId, Ordinal: ev.Ordinal, EventName: ev.Name, Err: tlErr})
   513			if h.projectionPolicies.Timeline == ProjectionErrorPolicyFail {
   514				return nil, tlErr
   515			}
   516		}
   517	
   518		entitiesToApply := entities
   519		if tlErr != nil {
   520			entitiesToApply = nil
   521		}
   522		if err := h.store.Apply(ctx, ev.SessionId, ev.Ordinal, entitiesToApply); err != nil {
   523			rec.AppliedEntities = cloneTimelineEntities(entitiesToApply)
   524			rec.ApplyErr = err
   525			h.reportError(ctx, ErrorRecord{Kind: ErrorKindStore, SessionId: ev.SessionId, Ordinal: ev.Ordinal, EventName: ev.Name, Err: err})
   526			return nil, err
   527		}
   528		rec.AppliedEntities = cloneTimelineEntities(entitiesToApply)
   529		if tlErr == nil {
   530			if projectionStore, ok := h.store.(ProjectionCursorStore); ok {
   531				if err := projectionStore.AdvanceProjectionCursor(ctx, TimelineProjectorName, ev.SessionId, ev.Ordinal); err != nil {
   532					rec.CursorErr = err
   533					h.reportError(ctx, ErrorRecord{Kind: ErrorKindStore, SessionId: ev.SessionId, Ordinal: ev.Ordinal, EventName: ev.Name, Err: err})
   534					return nil, err
   535				}
   536				rec.TimelineCursorAdvanced = true
   537			}
   538		}
   539		if uiErr == nil && h.fanout != nil && len(uiEvents) > 0 {
   540			fanoutEvents := cloneUIEvents(uiEvents)
   541			if err := h.fanout.PublishUI(ctx, ev.SessionId, ev.Ordinal, fanoutEvents); err != nil {
   542				rec.FanoutEvents = cloneUIEvents(fanoutEvents)
   543				rec.FanoutErr = err
   544				h.reportError(ctx, ErrorRecord{Kind: ErrorKindFanout, SessionId: ev.SessionId, Ordinal: ev.Ordinal, EventName: ev.Name, Err: err})
   545				return nil, err
   546			}
   547			rec.FanoutEvents = cloneUIEvents(fanoutEvents)
   548		}
   549		return uiEvents, nil
   550	}
   551	
   552	func cloneUIEvents(in []UIEvent) []UIEvent {
   553		if len(in) == 0 {
   554			return nil
   555		}
   556		out := make([]UIEvent, 0, len(in))
   557		for _, event := range in {
   558			clonedEvent := event
   559			if event.Payload != nil {
   560				clonedEvent.Payload = proto.Clone(event.Payload)
   561			}
   562			out = append(out, clonedEvent)
   563		}
   564		return out
   565	}

   567	func (h *Hub) nextLocalOrdinal(ctx context.Context, sid SessionId) (uint64, error) {
   568		h.mu.Lock()
   569		defer h.mu.Unlock()
   570		current, ok := h.localOrdinal[sid]
   571		if !ok {
   572			cursor, err := h.eventCursor(ctx, sid)
   573			if err != nil {
   574				return 0, err
   575			}
   576			current = cursor
   577		}
   578		current++
   579		h.localOrdinal[sid] = current
   580		return current, nil
   581	}

   583	func (h *Hub) eventCursor(ctx context.Context, sid SessionId) (uint64, error) {
   584		if eventStore, ok := h.store.(EventStore); ok {
   585			return eventStore.EventCursor(ctx, sid)
   586		}
   587		return h.store.Cursor(ctx, sid)
   588	}

   590	func (h *Hub) reportError(ctx context.Context, rec ErrorRecord) {
   591		if errorStore, ok := h.store.(ErrorStore); ok {
   592			if err := errorStore.RecordError(ctx, rec); err != nil {
   593				h.observeError(ctx, ErrorRecord{
   594					Kind:      ErrorKindStore,
   595					SessionId: rec.SessionId,
   596					Ordinal:   rec.Ordinal,
   597					EventName: rec.EventName,
   598					Err:       fmt.Errorf("record sessionstream error: %w", err),
   599					Metadata:  map[string]string{"originalKind": string(rec.Kind)},
   600				})
   601			}
   602		}
   603		h.observeError(ctx, rec)
   604	}

   606	func (h *Hub) observeError(ctx context.Context, rec ErrorRecord) {
   607		if h == nil || h.errorObserver == nil {
   608			return
   609		}
   610		safe := cloneErrorRecord(rec)
   611		defer func() { _ = recover() }()
   612		h.errorObserver.OnSessionstreamError(ctx, safe)
   613	}

   615	func cloneErrorRecord(in ErrorRecord) ErrorRecord {
   616		out := in
   617		if len(in.RawMessage) > 0 {
   618			out.RawMessage = append([]byte(nil), in.RawMessage...)
   619		}
   620		if len(in.Metadata) > 0 {
   621			out.Metadata = make(map[string]string, len(in.Metadata))
   622			for k, v := range in.Metadata {
   623				out.Metadata[k] = v
   624			}
   625		}
   626		return out
   627	}

   138	func (p watermillEventPublisher) Publish(ctx context.Context, ev Event) error {
   139		if p.hub == nil {
   140			return fmt.Errorf("hub is nil")
   141		}
   142		if p.hub.bus == nil {
   143			return fmt.Errorf("event bus is not configured")
   144		}
   145		if ev.SessionId == "" {
   146			return fmt.Errorf("event %q missing session id", ev.Name)
   147		}
   148		if ev.Payload == nil {
   149			return fmt.Errorf("event %q payload is nil", ev.Name)
   150		}
   151		if err := p.hub.validatePayloadType(p.hub.reg.events, "event", ev.Name, ev.Payload); err != nil {
   152			return err
   153		}
   154	
   155		payload, err := p.hub.reg.MarshalProtoJSON(ev.Payload)
   156		if err != nil {
   157			return fmt.Errorf("marshal event %q payload: %w", ev.Name, err)
   158		}
   159		body, err := json.Marshal(eventEnvelope{
   160			Name:      ev.Name,
   161			SessionID: string(ev.SessionId),
   162			Payload:   payload,
   163		})
   164		if err != nil {
   165			return fmt.Errorf("marshal event %q envelope: %w", ev.Name, err)
   166		}
   167	
   168		msg := message.NewMessage(uuid.NewString(), body)
   169		msg.SetContext(ctx)
   170		msg.Metadata.Set(MetadataKeyEventName, ev.Name)
   171		msg.Metadata.Set(MetadataKeySessionID, string(ev.SessionId))
   172		msg.Metadata.Set(MetadataKeyPartitionKey, PartitionKeyForSession(ev.SessionId))
   173		msg.Metadata.Set(MetadataKeyPublishedOrd, "0")
   174		if p.hub.bus.messageMutator != nil {
   175			if err := p.hub.bus.messageMutator(ctx, Event{Name: ev.Name, SessionId: ev.SessionId, Payload: proto.Clone(ev.Payload)}, msg); err != nil {
   176				return fmt.Errorf("mutate event %q message metadata: %w", ev.Name, err)
   177			}
   178		}
   179		if err := p.hub.bus.publisher.Publish(p.hub.bus.topic, msg); err != nil {
   180			return fmt.Errorf("publish event %q: %w", ev.Name, err)
   181		}
   182		if p.hub.bus.observer != nil {
   183			p.hub.bus.observer.Published(ctx, Event{Name: ev.Name, SessionId: ev.SessionId, Payload: proto.Clone(ev.Payload)}, newBusRecord(msg, p.hub.bus.topic))
   184		}
   185		return nil
   186	}

   188	func decodeEventEnvelope(reg *SchemaRegistry, payload []byte) (Event, error) {
   189		var env eventEnvelope
   190		if err := json.Unmarshal(payload, &env); err != nil {
   191			return Event{}, fmt.Errorf("decode event envelope: %w", err)
   192		}
   193		if env.Name == "" {
   194			return Event{}, fmt.Errorf("event envelope missing name")
   195		}
   196		if env.SessionID == "" {
   197			return Event{}, fmt.Errorf("event envelope missing session id")
   198		}
   199		msg, err := reg.instantiate(reg.events, "event", env.Name)
   200		if err != nil {
   201			return Event{}, err
   202		}
   203		if len(env.Payload) > 0 && string(env.Payload) != "null" {
   204			if err := protojson.Unmarshal(env.Payload, msg); err != nil {
   205				return Event{}, fmt.Errorf("decode event %q payload: %w", env.Name, err)
   206			}
   207		}
   208		return Event{Name: env.Name, SessionId: SessionId(env.SessionID), Payload: msg}, nil
   209	}

   155	// ServeHTTP upgrades a connection and serves the websocket protocol.
   156	func (s *Server) ServeHTTP(w http.ResponseWriter, r *http.Request) {
   157		conn, err := s.upgrader.Upgrade(w, r, nil)
   158		if err != nil {
   159			s.observe(r.Context(), TransportRecord{Stage: TransportStageUpgradeError, Err: err})
   160			return
   161		}
   162		cid := sessionstream.ConnectionId(fmt.Sprintf("conn-%d", atomic.AddUint64(&s.nextID, 1)))
   163		c := &connection{
   164			id:   cid,
   165			ws:   conn,
   166			send: make(chan outboundFrame, 128),
   167			subs: map[sessionstream.SessionId]subscription{},
   168		}
   169		s.mu.Lock()
   170		s.conns[cid] = c
   171		s.mu.Unlock()
   172		s.observe(r.Context(), TransportRecord{Stage: TransportStageConnected, ConnectionId: cid})
   173	
   174		go s.writeLoop(r.Context(), c)
   175		_ = s.sendFrame(c, newHelloFrame(cid))
   176		s.readLoop(r.Context(), c)
   177		s.closeConnection(c)
   178	}

   180	// PublishUI fans projected UI events out to subscribed websocket clients.
   181	func (s *Server) PublishUI(ctx context.Context, sid sessionstream.SessionId, ord uint64, events []sessionstream.UIEvent) error {
   182		if len(events) == 0 {
   183			return nil
   184		}
   185		targets := s.connectionsForSession(sid)
   186		targetIDs := fanoutTargetIDs(targets)
   187		if len(targets) == 0 {
   188			s.observe(ctx, TransportRecord{Stage: TransportStageFanoutNoTargets, SessionId: sid, Ordinal: ord, FanoutEventCount: len(events)})
   189			return nil
   190		}
   191		s.observe(ctx, TransportRecord{Stage: TransportStageFanoutStarted, SessionId: sid, Ordinal: ord, FanoutEventCount: len(events), FanoutTargetIds: targetIDs})
   192		deliveryErrs := make([]error, 0)
   193		for _, c := range targets {
   194			if err := s.deliverUIEvents(ctx, c, sid, ord, events); err != nil {
   195				deliveryErrs = append(deliveryErrs, fmt.Errorf("connection %s: %w", c.id, err))
   196				s.closeConnection(c)
   197			}
   198		}
   199		s.observe(ctx, TransportRecord{Stage: TransportStageFanoutCompleted, SessionId: sid, Ordinal: ord, FanoutEventCount: len(events), FanoutTargetIds: targetIDs, Err: errors.Join(deliveryErrs...)})
   200		return errors.Join(deliveryErrs...)
   201	}

   203	// Connections returns a stable snapshot of current connections and subscriptions.
   204	func (s *Server) Connections() []ConnectionInfo {
   205		s.mu.RLock()
   206		defer s.mu.RUnlock()
   207		out := make([]ConnectionInfo, 0, len(s.conns))
   208		for id, conn := range s.conns {
   209			conn.mu.RLock()
   210			subs := make([]string, 0, len(conn.subs))
   211			for sid := range conn.subs {
   212				subs = append(subs, string(sid))
   213			}
   214			conn.mu.RUnlock()
   215			sort.Strings(subs)
   216			out = append(out, ConnectionInfo{ConnectionId: string(id), Subscriptions: subs})
   217		}
   218		sort.Slice(out, func(i, j int) bool { return out[i].ConnectionId < out[j].ConnectionId })
   219		return out
   220	}

   222	func (s *Server) readLoop(ctx context.Context, c *connection) {
   223		for {
   224			select {
   225			case <-ctx.Done():
   226				return
   227			default:
   228			}
   229			_, raw, err := c.ws.ReadMessage()
   230			if err != nil {
   231				s.observe(ctx, TransportRecord{Stage: TransportStageReadError, Direction: FrameDirectionClientToServer, ConnectionId: c.id, Err: err})
   232				return
   233			}
   234			s.observe(ctx, TransportRecord{Stage: TransportStageClientFrameRead, Direction: FrameDirectionClientToServer, ConnectionId: c.id, RawBytes: len(raw)})
   235			frame := &sessionstreamv1.ClientFrame{}
   236			if err := unmarshalOpts.Unmarshal(raw, frame); err != nil {
   237				s.observe(ctx, TransportRecord{Stage: TransportStageClientFrameDecodeError, Direction: FrameDirectionClientToServer, ConnectionId: c.id, RawBytes: len(raw), Err: err})
   238				_ = s.sendFrame(c, newErrorFrame("bad_client_frame", err.Error(), ""))
   239				continue
   240			}
   241			s.observe(ctx, clientFrameRecord(TransportStageClientFrameDecoded, c.id, frame, len(raw)))
   242			if err := s.handleClientFrame(ctx, c, frame); err != nil {
   243				rec := clientFrameRecord(TransportStageProtocolError, c.id, frame, 0)
   244				rec.Err = err
   245				s.observe(ctx, rec)
   246				_ = s.sendFrame(c, newErrorFrame("protocol_error", err.Error(), ""))
   247			}
   248		}
   249	}

   251	func (s *Server) handleClientFrame(ctx context.Context, c *connection, frame *sessionstreamv1.ClientFrame) error {
   252		switch typed := frame.GetFrame().(type) {
   253		case *sessionstreamv1.ClientFrame_Ping:
   254			return s.sendFrame(c, newPongFrame(typed.Ping.GetNonce()))
   255		case *sessionstreamv1.ClientFrame_Pong:
   256			return nil
   257		case *sessionstreamv1.ClientFrame_Subscribe:
   258			sub := typed.Subscribe
   259			sid := sessionstream.SessionId(sub.GetSessionId())
   260			if sid == "" {
   261				return fmt.Errorf("subscribe missing session id")
   262			}
   263			since := sub.GetSinceSnapshotOrdinal()
   264			s.observe(ctx, TransportRecord{Stage: TransportStageSubscribeReceived, Direction: FrameDirectionClientToServer, ConnectionId: c.id, SessionId: sid, FrameType: "subscribe", SinceSnapshotOrdinal: since})
   265			s.registerHydrating(c, sid, since)
   266			completed := false
   267			defer func() {
   268				if !completed {
   269					s.removeSubscription(c, sid)
   270				}
   271			}()
   272	
   273			s.observe(ctx, TransportRecord{Stage: TransportStageSnapshotLoadStarted, ConnectionId: c.id, SessionId: sid, SinceSnapshotOrdinal: since})
   274			snap, err := s.snapshots.Snapshot(ctx, sid)
   275			if err != nil {
   276				s.observe(ctx, TransportRecord{Stage: TransportStageProtocolError, ConnectionId: c.id, SessionId: sid, SinceSnapshotOrdinal: since, Err: err})
   277				return fmt.Errorf("load snapshot for %q: %w", sid, err)
   278			}
   279			snapshotRecord := TransportRecord{Stage: TransportStageSnapshotLoaded, ConnectionId: c.id, SessionId: sid, SinceSnapshotOrdinal: since, SnapshotOrdinal: snap.SnapshotOrdinal, SnapshotEntityCount: len(snap.Entities), SnapshotEntities: summarizeEntities(snap.Entities)}
   280			s.observe(ctx, snapshotRecord)
   281			if err := s.sendFrame(c, newSnapshotFrame(sid, snap)); err != nil {
   282				return err
   283			}
   284			snapshotRecord.Stage = TransportStageSnapshotSent
   285			s.observe(ctx, snapshotRecord)
   286			buffered := s.drainHydrationBuffer(c, sid, snap.SnapshotOrdinal)
   287			if len(buffered) > 0 {
   288				s.observe(ctx, TransportRecord{Stage: TransportStageHydrationBufferFlushed, ConnectionId: c.id, SessionId: sid, SnapshotOrdinal: snap.SnapshotOrdinal, FanoutEventCount: countBufferedEvents(buffered)})
   289			}
   290			for _, batch := range buffered {
   291				if err := s.sendUIBatch(ctx, c, sid, batch.ordinal, batch.events); err != nil {
   292					return err
   293				}
   294			}
   295			lateEventCount, err := s.flushLateHydrationBufferAndMarkLive(ctx, c, sid, snap.SnapshotOrdinal)
   296			if err != nil {
   297				return err
   298			}
   299			if lateEventCount > 0 {
   300				s.observe(ctx, TransportRecord{Stage: TransportStageHydrationBufferFlushed, ConnectionId: c.id, SessionId: sid, SnapshotOrdinal: snap.SnapshotOrdinal, FanoutEventCount: lateEventCount})
   301			}
   302			if err := s.sendFrame(c, newSubscribedFrame(sid, since)); err != nil {
   303				return err
   304			}
   305			s.observe(ctx, TransportRecord{Stage: TransportStageSubscribed, ConnectionId: c.id, SessionId: sid, SinceSnapshotOrdinal: since, SnapshotOrdinal: snap.SnapshotOrdinal})
   306			completed = true
   307			return nil
   308		case *sessionstreamv1.ClientFrame_Unsubscribe:
   309			sid := sessionstream.SessionId(typed.Unsubscribe.GetSessionId())
   310			if sid == "" {
   311				return fmt.Errorf("unsubscribe missing session id")
   312			}
   313			s.observe(ctx, TransportRecord{Stage: TransportStageUnsubscribeReceived, Direction: FrameDirectionClientToServer, ConnectionId: c.id, SessionId: sid, FrameType: "unsubscribe"})
   314			s.removeSubscription(c, sid)
   315			if err := s.sendFrame(c, newUnsubscribedFrame(sid)); err != nil {
   316				return err
   317			}
   318			s.observe(ctx, TransportRecord{Stage: TransportStageUnsubscribed, ConnectionId: c.id, SessionId: sid})
   319			return nil
   320		default:
   321			return fmt.Errorf("unknown client frame")
   322		}
   323	}

   325	func (s *Server) writeLoop(ctx context.Context, c *connection) {
   326		for msg := range c.send {
   327			if err := c.ws.WriteMessage(websocket.TextMessage, msg.body); err != nil {
   328				s.observe(ctx, TransportRecord{Stage: TransportStageServerFrameWriteError, Direction: FrameDirectionServerToClient, ConnectionId: c.id, FrameType: msg.frameType, RawBytes: len(msg.body), Err: err})
   329				return
   330			}
   331			s.observe(ctx, TransportRecord{Stage: TransportStageServerFrameWritten, Direction: FrameDirectionServerToClient, ConnectionId: c.id, FrameType: msg.frameType, RawBytes: len(msg.body)})
   332		}
   333	}

   335	func (s *Server) closeConnection(c *connection) {
   336		if c == nil {
   337			return
   338		}
   339		c.close.Do(func() {
   340			c.mu.Lock()
   341			subs := make([]sessionstream.SessionId, 0, len(c.subs))
   342			for sid := range c.subs {
   343				subs = append(subs, sid)
   344			}
   345			c.subs = map[sessionstream.SessionId]subscription{}
   346			c.mu.Unlock()
   347	
   348			s.mu.Lock()
   349			delete(s.conns, c.id)
   350			for _, sid := range subs {
   351				delete(s.bySession[sid], c.id)
   352				if len(s.bySession[sid]) == 0 {
   353					delete(s.bySession, sid)
   354				}
   355			}
   356			s.mu.Unlock()
   357	
   358			c.closed.Store(true)
   359			close(c.send)
   360			_ = c.ws.Close()
   361			s.observe(context.Background(), TransportRecord{Stage: TransportStageDisconnected, ConnectionId: c.id})
   362		})
   363	}

   381	func (s *Server) registerHydrating(c *connection, sid sessionstream.SessionId, since uint64) {
   382		c.mu.Lock()
   383		c.subs[sid] = subscription{sinceSnapshotOrdinal: since, state: subscriptionStateHydrating}
   384		c.mu.Unlock()
   385	
   386		s.mu.Lock()

   388		if set == nil {
   389			set = map[sessionstream.ConnectionId]struct{}{}

   397	// deliverUIEvents routes a UI event batch to one connection. It checks the
   398	// subscription state under the connection lock and either buffers (hydrating)
   399	// or sends directly (live). This eliminates the TOCTOU race where PublishUI
   400	// reads a stale subscription state while another goroutine transitions the
   401	// subscription between hydration and live delivery.
   402	func (s *Server) deliverUIEvents(ctx context.Context, c *connection, sid sessionstream.SessionId, ord uint64, events []sessionstream.UIEvent) error {
   403		c.mu.Lock()
   404		sub, ok := c.subs[sid]
   405		if !ok {
   406			c.mu.Unlock()
   407			return nil
   408		}
   409		switch sub.state {
   410		case subscriptionStateLive:
   411			c.mu.Unlock()
   412			return s.sendUIBatch(ctx, c, sid, ord, events)
   413		case subscriptionStateHydrating:
   414			if len(sub.buffer) >= s.maxHydrationBufferedBatches {
   415				err := fmt.Errorf("connection %s hydration buffer full for session %s", c.id, sid)
   416				c.mu.Unlock()
   417				s.observe(ctx, TransportRecord{Stage: TransportStageHydrationBufferOverflow, ConnectionId: c.id, SessionId: sid, Ordinal: ord, FanoutEventCount: len(events), Err: err})
   418				_ = s.sendFrame(c, newErrorFrame("hydration_buffer_overflow", err.Error(), ""))
   419				return err
   420			}
   421			sub.buffer = append(sub.buffer, bufferedUIBatch{ordinal: ord, events: cloneUIEvents(events)})
   422			c.subs[sid] = sub
   423			c.mu.Unlock()
   424			s.observe(ctx, TransportRecord{Stage: TransportStageUIEventBuffered, ConnectionId: c.id, SessionId: sid, Ordinal: ord, FanoutEventCount: len(events)})
   425			return nil
   426		default:
   427			c.mu.Unlock()
   428			return nil
   429		}
   430	}

   432	func (s *Server) drainHydrationBuffer(c *connection, sid sessionstream.SessionId, snapshotOrdinal uint64) []bufferedUIBatch {
   433		c.mu.Lock()
   434		defer c.mu.Unlock()
   435		sub, ok := c.subs[sid]
   436		if !ok {
   437			return nil
   438		}
   439		out := filterBufferedAfterSnapshot(sub.buffer, snapshotOrdinal)
   440		sub.buffer = nil
   441		sub.snapshotOrdinal = snapshotOrdinal
   442		c.subs[sid] = sub
   443		return out
   444	}

   446	// flushLateHydrationBufferAndMarkLive transitions a subscription from hydrating
   447	// to live only after any batches buffered since drainHydrationBuffer have been
   448	// queued to the connection. Holding c.mu while queueing those late batches keeps
   449	// concurrent PublishUI calls from observing the live state and enqueueing newer
   450	// events ahead of older late-hydration events.
   451	func (s *Server) flushLateHydrationBufferAndMarkLive(ctx context.Context, c *connection, sid sessionstream.SessionId, snapshotOrdinal uint64) (int, error) {
   452		c.mu.Lock()
   453		defer c.mu.Unlock()
   454		sub, ok := c.subs[sid]
   455		if !ok {
   456			return 0, nil
   457		}
   458		late := filterBufferedAfterSnapshot(sub.buffer, snapshotOrdinal)
   459		for _, batch := range late {
   460			if err := s.sendUIBatch(ctx, c, sid, batch.ordinal, batch.events); err != nil {
   461				return 0, err
   462			}
   463		}
   464		sub.state = subscriptionStateLive
   465		sub.snapshotOrdinal = snapshotOrdinal
   466		sub.buffer = nil
   467		c.subs[sid] = sub
   468		s.observe(context.Background(), TransportRecord{Stage: TransportStageSubscriptionLive, ConnectionId: c.id, SessionId: sid, SinceSnapshotOrdinal: sub.sinceSnapshotOrdinal, SnapshotOrdinal: snapshotOrdinal})
   469		return countBufferedEvents(late), nil
   470	}

   472	func (s *Server) sendUIBatch(ctx context.Context, c *connection, sid sessionstream.SessionId, ord uint64, events []sessionstream.UIEvent) error {
   473		for _, event := range events {
   474			frame, err := newUIEventFrame(sid, ord, event)

   482		}
   483		return nil
   484	}

   486	func filterBufferedAfterSnapshot(batches []bufferedUIBatch, snapshotOrdinal uint64) []bufferedUIBatch {
   487		out := make([]bufferedUIBatch, 0, len(batches))
   488		for _, batch := range batches {
   489			if batch.ordinal > snapshotOrdinal {
   490				out = append(out, cloneBufferedUIBatch(batch))
   491			}
   492		}
   493		sort.SliceStable(out, func(i, j int) bool { return out[i].ordinal < out[j].ordinal })
   494		return out
   495	}

   524	func (s *Server) sendFrame(c *connection, frame *sessionstreamv1.ServerFrame) (err error) {
   525		frameType := serverFrameType(frame)
   526		if c == nil {
   527			return fmt.Errorf("connection is nil")
   528		}
   529		if c.closed.Load() {
   530			return fmt.Errorf("connection %s is closed", c.id)
   531		}
   532		body, err := marshalOptions.Marshal(frame)
   533		if err != nil {
   534			s.observe(context.Background(), TransportRecord{Stage: TransportStageServerFrameMarshalError, Direction: FrameDirectionServerToClient, ConnectionId: c.id, FrameType: frameType, Err: err})
   535			return err
   536		}
   537		defer func() {
   538			if r := recover(); r != nil {
   539				err = fmt.Errorf("connection %s is closed", c.id)
   540			}
   541		}()
   542		queueLen := len(c.send)
   543		queueCap := cap(c.send)
   544		select {
   545		case c.send <- outboundFrame{body: body, frameType: frameType}:
   546			s.observe(context.Background(), TransportRecord{Stage: TransportStageServerFrameQueued, Direction: FrameDirectionServerToClient, ConnectionId: c.id, FrameType: frameType, RawBytes: len(body), QueueLen: queueLen, QueueCap: queueCap})
   547			return nil
   548		default:
   549			if c.closed.Load() {
   550				return fmt.Errorf("connection %s is closed", c.id)
   551			}
   552			err := fmt.Errorf("connection %s send buffer full", c.id)
   553			s.observe(context.Background(), TransportRecord{Stage: TransportStageServerFrameQueueFull, Direction: FrameDirectionServerToClient, ConnectionId: c.id, FrameType: frameType, RawBytes: len(body), QueueLen: queueLen, QueueCap: queueCap, Err: err})
   554			return err
   555		}
   556	}

   104		if f != nil {

   113			return nil
   114		}

   117	func (s *Server) observe(ctx context.Context, rec TransportRecord) {
   118		if s == nil || s.observer == nil {
   119			return
   120		}
   121		safe := cloneTransportRecord(rec)
   122		defer func() { _ = recover() }()
   123		s.observer.OnTransport(ctx, safe)
   124	}

    11	type Filter struct {
    12		WorkflowID string
    13		OpID       string
    14		Site       string
    15		WorkerID   string
    16		Limit      int
    17	}
    18	
    19	type Hub struct {
    20		mu          sync.RWMutex
    21		capacity    int
    22		events      []*runtimev1.RuntimeEventV1
    23		subscribers map[chan *runtimev1.RuntimeEventV1]Filter
    24	}
    25	
    26	func NewHub(capacity int) *Hub {
    27		if capacity <= 0 {
    28			capacity = DefaultRecentEventLimit
    29		}
    30		return &Hub{
    31			capacity:    capacity,
    32			subscribers: map[chan *runtimev1.RuntimeEventV1]Filter{},
    33		}
    34	}
    35	
    36	func (h *Hub) Add(event *runtimev1.RuntimeEventV1) {
    37		if h == nil || event == nil {
    38			return
    39		}
    40	
    41		cloned := proto.Clone(event).(*runtimev1.RuntimeEventV1)
    42	
    43		h.mu.Lock()
    44		h.events = append(h.events, cloned)
    45		if len(h.events) > h.capacity {
    46			h.events = append([]*runtimev1.RuntimeEventV1(nil), h.events[len(h.events)-h.capacity:]...)
    47		}
    48	
    49		deliveries := make([]struct {
    50			ch    chan *runtimev1.RuntimeEventV1
    51			event *runtimev1.RuntimeEventV1
    52		}, 0, len(h.subscribers))
    53		for ch, filter := range h.subscribers {
    54			if !matchesFilter(cloned, filter) {
    55				continue
    56			}
    57			deliveries = append(deliveries, struct {
    58				ch    chan *runtimev1.RuntimeEventV1
    59				event *runtimev1.RuntimeEventV1
    60			}{
    61				ch:    ch,
    62				event: proto.Clone(cloned).(*runtimev1.RuntimeEventV1),
    63			})
    64		}
    65		h.mu.Unlock()
    66	
    67		for _, delivery := range deliveries {
    68			select {
    69			case delivery.ch <- delivery.event:
    70			default:
    71			}
    72		}
    73	}
    74	
    75	func (h *Hub) Recent(filter Filter) []*runtimev1.RuntimeEventV1 {
    76		if h == nil {
    77			return nil
    78		}
    79	
    80		h.mu.RLock()
    81		defer h.mu.RUnlock()
    82	
    83		matches := make([]*runtimev1.RuntimeEventV1, 0, len(h.events))
    84		for _, event := range h.events {
    85			if !matchesFilter(event, filter) {
    86				continue
    87			}
    88			matches = append(matches, proto.Clone(event).(*runtimev1.RuntimeEventV1))
    89		}
    90	
    91		if filter.Limit > 0 && len(matches) > filter.Limit {
    92			matches = matches[len(matches)-filter.Limit:]
    93		}
    94		return matches
    95	}
    96	
    97	func (h *Hub) Subscribe(ctx context.Context, filter Filter) <-chan *runtimev1.RuntimeEventV1 {
    98		if h == nil {
    99			closed := make(chan *runtimev1.RuntimeEventV1)
   100			close(closed)
   101			return closed
   102		}
   103	
   104		ch := make(chan *runtimev1.RuntimeEventV1, 32)
   105		h.mu.Lock()
   106		h.subscribers[ch] = filter
   107		h.mu.Unlock()
   108	
   109		go func() {
   110			<-ctx.Done()
   111			h.mu.Lock()
   112			delete(h.subscribers, ch)
   113			close(ch)
   114			h.mu.Unlock()
   115		}()
   116	
   117		return ch
   118	}
   119	
   120	func matchesFilter(event *runtimev1.RuntimeEventV1, filter Filter) bool {
   121		if event == nil {
   122			return false
   123		}
   124		if filter.WorkflowID != "" && event.WorkflowId != filter.WorkflowID {
   125			return false
   126		}
   127		if filter.OpID != "" && event.OpId != filter.OpID {
   128			return false
   129		}
   130		if filter.Site != "" && event.Site != filter.Site {
   131			return false
   132		}
   133		if filter.WorkerID != "" && event.WorkerId != filter.WorkerID {
   134			return false
   135		}
   136		return true
   137	}

    91	func open(cfg Config, withSubscriber bool) (*Resources, error) {
    92		cfg = cfg.Normalize()
    93		if err := cfg.Validate(); err != nil {
    94			return nil, err
    95		}
    96	
    97		switch cfg.Backend {
    98		case BackendOff:
    99			return &Resources{Topic: cfg.Topic, Publisher: nopPublisher{}}, nil
   100		case BackendGoChannel:
   101			ch := cfg.GoChannel
   102			if ch == nil {
   103				ch = NewGoChannelPubSub()
   104			}
   105			res := &Resources{
   106				Topic:     cfg.Topic,
   107				Publisher: ch,
   108				closers:   []func() error{ch.Close},
   109			}
   110			if withSubscriber {
   111				res.Subscriber = ch
   112			}
   113			return res, nil
   114		case BackendRedis:
   115			return openRedis(cfg, withSubscriber)
   116		default:
   117			return nil, fmt.Errorf("unsupported runtime events backend %q", cfg.Backend)
   118		}
   119	}

   121	func openRedis(cfg Config, withSubscriber bool) (*Resources, error) {
   122		logger := watermill.NopLogger{}
   123		publisherClient := redis.NewClient(&redis.Options{
   124			Addr:     cfg.RedisAddress,
   125			Username: cfg.RedisUsername,
   126			Password: cfg.RedisPassword,
   127			DB:       cfg.RedisDB,
   128		})
   129	
   130		publisher, err := redisstream.NewPublisher(redisstream.PublisherConfig{
   131			Client:        publisherClient,
   132			DefaultMaxlen: cfg.RedisStreamMaxLen,
   133		}, logger)
   134		if err != nil {
   135			_ = publisherClient.Close()
   136			return nil, fmt.Errorf("open redis runtime-event publisher: %w", err)
   137		}
   138	
   139		res := &Resources{
   140			Topic:     cfg.Topic,
   141			Publisher: publisher,
   142			closers:   []func() error{publisher.Close},
   143		}
   144		if !withSubscriber {
   145			return res, nil
   146		}
   147	
   148		subscriberClient := redis.NewClient(&redis.Options{
   149			Addr:     cfg.RedisAddress,
   150			Username: cfg.RedisUsername,
   151			Password: cfg.RedisPassword,
   152			DB:       cfg.RedisDB,
   153		})
   154	
   155		subscriber, err := redisstream.NewSubscriber(redisstream.SubscriberConfig{
   156			Client:        subscriberClient,
   157			Consumer:      cfg.RedisConsumer,
   158			ConsumerGroup: cfg.RedisConsumerGroup,
   159			OldestId:      "$",
   160		}, logger)
   161		if err != nil {
   162			_ = publisher.Close()
   163			_ = subscriberClient.Close()
   164			return nil, fmt.Errorf("open redis runtime-event subscriber: %w", err)
   165		}
   166	
   167		res.Subscriber = subscriber
   168		res.closers = append(res.closers, subscriber.Close)
   169		return res, nil
   170	}

   186	func (r *Resources) Close() error {
   187		if r == nil {
   188			return nil
   189		}
   190	
   191		var firstErr error
   192		for i := len(r.closers) - 1; i >= 0; i-- {
   193			if err := r.closers[i](); err != nil && firstErr == nil {
   194				firstErr = err
   195			}
   196		}
   197		r.closers = nil
   198		return firstErr
   199	}

    38	func (r *ObservedRunner) Run(ctx context.Context, runCtx runner.RunContext) (*model.OpResult, error) {
    39		startedAt := time.Now().UTC()
    40		_ = r.publisher.Publish(buildRunnerLogEvent(
    41			runCtx,
    42			r.component,
    43			r.workerID,
    44			runtimev1.RuntimeEventSeverity_RUNTIME_EVENT_SEVERITY_INFO,

    53		result, err := r.base.Run(ctx, runCtx)
    54		completedAt := time.Now().UTC()
    55	
    56		payload := map[string]any{
    57			"runnerKind":        r.base.Kind(),
    58			"durationMillis":    completedAt.Sub(startedAt).Milliseconds(),
    59			"emittedCount":      0,
    60			"artifactCount":     0,
    61			"recordWriteCount":  0,
    62			"artifactSummaries": []any{},
    63		}
    64		severity := runtimev1.RuntimeEventSeverity_RUNTIME_EVENT_SEVERITY_INFO
    65		message := "runner completed"

    68			severity = runtimev1.RuntimeEventSeverity_RUNTIME_EVENT_SEVERITY_ERROR
    69			message = "runner failed"
    70			errorCode = "runner_error"
    71			payload["error"] = err.Error()
    72		} else if result != nil {
    73			payload["emittedCount"] = len(result.Emitted)
    74			payload["artifactCount"] = len(result.Artifacts)
    75			payload["recordWriteCount"] = len(result.Records)
    76			payload["artifactSummaries"] = artifactSummaries(result.Artifacts)
    77			if result.Error != nil {
    78				severity = runtimev1.RuntimeEventSeverity_RUNTIME_EVENT_SEVERITY_ERROR
    79				message = "runner returned op error"
    80				errorCode = result.Error.Code
    81				payload["error"] = result.Error.Message
    82				payload["retryable"] = result.Error.Retryable
    83			}
    84		}
    85	
    86		_ = r.publisher.Publish(buildRunnerLogEvent(
    87			runCtx,
    88			r.component,
    89			r.workerID,
    90			severity,
    91			message,
    92			completedAt,
    93			payload,
    94			errorCode,

    21	func (h *RuntimeEventsHandler) List(w http.ResponseWriter, r *http.Request) {
    22		events := h.hub.Recent(runtimeEventFilterFromRequest(r))
    23		rawEvents := make([]json.RawMessage, 0, len(events))
    24		for _, event := range events {
    25			raw, err := runtimeevents.MarshalJSON(event)
    26			if err != nil {
    27				writeError(w, http.StatusInternalServerError, "event_encode_failed", "failed to encode runtime events")
    28				return
    29			}
    30			rawEvents = append(rawEvents, raw)
    31		}
    32		writeJSON(w, http.StatusOK, map[string]any{"events": rawEvents})
    33	}
    34	
    35	func (h *RuntimeEventsHandler) Stream(w http.ResponseWriter, r *http.Request) {
    36		flusher, ok := w.(http.Flusher)
    37		if !ok {
    38			writeError(w, http.StatusInternalServerError, "stream_unsupported", "streaming is not supported")
    39			return
    40		}
    41	
    42		w.Header().Set("Content-Type", "text/event-stream")
    43		w.Header().Set("Cache-Control", "no-cache")
    44		w.Header().Set("Connection", "keep-alive")
    45		w.Header().Set("X-Accel-Buffering", "no")
    46	
    47		stream := h.hub.Subscribe(r.Context(), runtimeEventFilterFromRequest(r))
    48		heartbeat := time.NewTicker(15 * time.Second)
    49		defer heartbeat.Stop()
    50	
    51		flusher.Flush()
    52		for {
    53			select {
    54			case <-r.Context().Done():
    55				return
    56			case <-heartbeat.C:
    57				_, _ = fmt.Fprint(w, ": ping\n\n")
    58				flusher.Flush()
    59			case event, ok := <-stream:
    60				if !ok {
    61					return
    62				}
    63				raw, err := runtimeevents.MarshalJSON(event)
    64				if err != nil {
    65					continue
    66				}
    67				if event.Id != "" {
    68					_, _ = fmt.Fprintf(w, "id: %s\n", event.Id)
    69				}
    70				_, _ = fmt.Fprint(w, "event: runtime-event\n")
    71				_, _ = fmt.Fprintf(w, "data: %s\n\n", raw)
    72				flusher.Flush()
    73			}
    74		}
    75	}

    82			WorkerID:   r.URL.Query().Get("workerId"),

    86				filter.Limit = limit

    63	const MAX_CACHED_EVENTS = 500;
    64	
    65	function isStorybookEnvironment(): boolean {
    66	  return (
    67	    typeof window !== 'undefined' &&
    68	    '__STORYBOOK_PREVIEW__' in window
    69	  );

    72	// ── API ──────────────────────────────────────────────────────────────
    73	
    74	export const runtimeEventsApi = createApi({
    75	  reducerPath: 'runtimeEventsApi',
    76	  baseQuery: fetchBaseQuery({ baseUrl: '/api/v1' }),
    77	  tagTypes: ['RuntimeEvents'],
    78	  endpoints: (builder) => ({
    79	    getRecentRuntimeEvents: builder.query<RuntimeEventJson[], RuntimeEventsParams>({
    80	      query: (params) => buildRuntimeEventQuery(params),
    81	      transformResponse: (response: RuntimeEventsResponse) => response.events,
    82	      providesTags: ['RuntimeEvents'],
    83	      keepUnusedDataFor: 30,
    84	
    85	      async onCacheEntryAdded(
    86	        arg,
    87	        { updateCachedData, cacheDataLoaded, cacheEntryRemoved },
    88	      ) {
    89	        // Wait for the initial REST fetch to resolve.
    90	        // If there is no backend, cacheDataLoaded rejects and we return early —
    91	        // no SSE connection is opened, no infinite loop.
    92	        try {
    93	          await cacheDataLoaded;
    94	        } catch {
    95	          return;
    96	        }
    97	
    98	        // Storybook only needs the mocked REST payload. Skip live SSE there to
    99	        // avoid noisy 404s from the unmocked stream endpoint.
   100	        if (isStorybookEnvironment()) {
   101	          await cacheEntryRemoved;
   102	          return;
   103	        }
   104	
   105	        // Open SSE stream with the same scoping filters
   106	        const sseUrl = buildSSEUrl(arg);
   107	        const eventSource = new EventSource(sseUrl);
   108	
   109	        const onMessage = (event: MessageEvent<string>) => {
   110	          try {
   111	            const decoded = JSON.parse(event.data) as RuntimeEventJson;
   112	            updateCachedData((draft) => {
   113	              // Dedupe
   114	              const exists = draft.some((e) => {
   115	                if (!e || typeof e !== 'object' || !decoded || typeof decoded !== 'object') {
   116	                  return false;
   117	                }
   118	                return 'id' in e && 'id' in decoded && e.id === decoded.id;
   119	              });
   120	              if (!exists) draft.unshift(decoded);
   121	              // Trim to max
   122	              if (draft.length > MAX_CACHED_EVENTS) {
   123	                draft.length = MAX_CACHED_EVENTS;
   124	              }
   125	            });
   126	          } catch {
   127	            // ignore malformed event payloads
   128	          }
   129	        };
   130	
   131	        eventSource.addEventListener('runtime-event', onMessage as EventListener);
   132	
   133	        // Close SSE on error to prevent browser auto-reconnect
   134	        eventSource.addEventListener('error', () => {
   135	          eventSource.close();
   136	        });
   137	
   138	        // Auto-cleanup when no subscribers remain
   139	        await cacheEntryRemoved;
   140	        eventSource.close();
   141	      },
   142	    }),
   143	  }),
   144	});
